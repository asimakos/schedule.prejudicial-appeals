
CREATE TABLE appeals (
    id int generated by default as identity,
    name varchar(255),
    date_entry date,
    date_suspended date,
    date_court date,
    date_judgement date,
    promoter_id int DEFAULT NULL,
    CONSTRAINT PK_appeals PRIMARY KEY (id),
    CONSTRAINT FK_appealspromoter 
    FOREIGN KEY (promoter_id) REFERENCES promoter(id)
);


CREATE TABLE promoter (
    id int generated by default as identity,
    fullname varchar(255),
    CONSTRAINT PK_promoter PRIMARY KEY (id)
);
